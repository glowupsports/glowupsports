📁 scheduling/availability/availability.service.ts
import { db } from "@/shared/db";

export class AvailabilityService {

  async lockSlot(
    tx: any,
    slotId: string,
    lockMinutes = 5
  ) {
    const slot = await tx.timeSlots.findUnique({
      where: { id: slotId },
      lock: { mode: "for update" }
    });

    if (!slot || slot.status !== "available") {
      throw new Error("Slot not available");
    }

    await tx.timeSlots.update({
      where: { id: slotId },
      data: {
        status: "locked",
        lockedUntil: new Date(Date.now() + lockMinutes * 60_000)
      }
    });
  }

  async confirmSlot(tx: any, slotId: string) {
    await tx.timeSlots.update({
      where: { id: slotId },
      data: {
        status: "booked",
        lockedUntil: null
      }
    });
  }

  async releaseExpiredLocks(tx: any) {
    await tx.timeSlots.updateMany({
      where: {
        status: "locked",
        lockedUntil: { lt: new Date() }
      },
      data: {
        status: "available",
        lockedUntil: null
      }
    });
  }
}

📁 scheduling/bookings/booking.service.ts
import { db } from "@/shared/db";
import { AvailabilityService } from "../availability/availability.service";

export class BookingService {
  private availability = new AvailabilityService();

  async createBookingRequest(input: {
    playerId: string;
    slotId: string;
    price: number;
  }) {
    return db.transaction(async tx => {
      await this.availability.lockSlot(tx, input.slotId);

      return tx.bookingRequests.create({
        data: {
          playerId: input.playerId,
          slotId: input.slotId,
          priceSnapshot: input.price,
          status: "pending"
        }
      });
    });
  }

  async approveBooking(bookingId: string, adminId: string) {
    return db.transaction(async tx => {
      const booking = await tx.bookingRequests.findUnique({
        where: { id: bookingId },
        lock: { mode: "for update" }
      });

      if (!booking || booking.status !== "pending") {
        throw new Error("Invalid booking state");
      }

      await this.availability.confirmSlot(tx, booking.slotId);

      await tx.sessions.create({
        data: {
          bookingId: booking.id,
          playerId: booking.playerId,
          startTime: booking.startTime,
          endTime: booking.endTime
        }
      });

      await tx.bookingRequests.update({
        where: { id: bookingId },
        data: { status: "approved" }
      });
    });
  }
}

✅ PART 4 — TIMESLOT LOCKING MODEL (PLATFORM INVARIANT)

This is non-negotiable for your platform.

Slot States
AVAILABLE → LOCKED → BOOKED

Rules

Lock expires automatically

Only ONE transaction can lock a slot

Booking approval confirms slot

Cancellation returns slot to AVAILABLE

Locks use SELECT … FOR UPDATE

This model:

works on Postgres / Supabase

works with mobile

prevents race conditions

is simple enough to reason about
