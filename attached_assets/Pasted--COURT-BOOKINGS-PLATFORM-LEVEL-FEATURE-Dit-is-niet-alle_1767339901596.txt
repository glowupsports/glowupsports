Copy code

  id
  academy_id
  name
  is_public
  price_per_hour
}

court_availability {
  court_id
  date
  time_slot
  status: available | booked | blocked
}

court_bookings {
  court_id
  user_id
  start_time
  end_time
  type: public | academy | training
  price
}
