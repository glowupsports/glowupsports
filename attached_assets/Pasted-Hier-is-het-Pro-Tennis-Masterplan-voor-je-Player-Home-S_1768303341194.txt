import { LinearGradient } from 'expo-linear-gradient';

import { Text, TouchableOpacity, StyleSheet } from 'react-native';

import { PRO_TENNIS_COLORS } from '@/constants/Colors';



export const ProButton = ({ title, onPress }) => (

  <TouchableOpacity onPress={onPress}>

    <LinearGradient

      // De 'Electric' overgang van groen naar iets lichter groen

      colors={[PRO_TENNIS_COLORS.electricGreen, '#DFFF40']}

      start={{ x: 0, y: 0 }}

      end={{ x: 1, y: 1 }}

      style={styles.button}

    >

      <Text style={styles.text}>{title.toUpperCase()}</Text>

    </LinearGradient>

  </TouchableOpacity>

);



const styles = StyleSheet.create({

  button: {

    paddingVertical: 16,

    paddingHorizontal: 32,

    borderRadius: 8, // Iets ronder voor de 'Sport' look

    alignItems: 'center',

    shadowColor: PRO_TENNIS_COLORS.electricGreen,

    shadowOffset: { width: 0, height: 4 },

    shadowOpacity: 0.3,

    shadowRadius: 10,

    elevation: 5,

  },

  text: {

    color: '#000', // Zwart op Geel/Groen leest het best (Nike stijl)

    fontWeight: '900',

    fontSize: 16,

    letterSpacing: 1,

  }

});
