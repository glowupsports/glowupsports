4. GlowRing.tsx
import React, { useEffect } from "react";
import { StyleSheet, View } from "react-native";
import Svg, { Circle, Defs, LinearGradient, Stop } from "react-native-svg";
import Animated, {
  Easing,
  useAnimatedStyle,
  useSharedValue,
  withRepeat,
  withTiming,
} from "react-native-reanimated";
import { GLOW_COLORS } from "./glowTokens";

const AnimatedView = Animated.createAnimatedComponent(View);

type GlowRingProps = {
  size?: number;
  strokeWidth?: number;
};

export default function GlowRing({
  size = 280,
  strokeWidth = 4,
}: GlowRingProps) {
  const rotation = useSharedValue(0);

  useEffect(() => {
    rotation.value = withRepeat(
      withTiming(360, {
        duration: 10000,
        easing: Easing.linear,
      }),
      -1,
      false
    );
  }, [rotation]);

  const animatedStyle = useAnimatedStyle(() => ({
    transform: [{ rotate: `${rotation.value}deg` }],
  }));

  const radius = (size - strokeWidth) / 2;

  return (
    <AnimatedView style={[styles.wrapper, { width: size, height: size }, animatedStyle]}>
      <View
        style={[
          styles.outerGlow,
          {
            width: size,
            height: size,
            borderRadius: size / 2,
            shadowColor: GLOW_COLORS.green,
          },
        ]}
      />
      <Svg width={size} height={size}>
        <Defs>
          <LinearGradient id="ringGradient" x1="0%" y1="0%" x2="100%" y2="100%">
            <Stop offset="0%" stopColor={GLOW_COLORS.purple} />
            <Stop offset="55%" stopColor={GLOW_COLORS.cyan} />
            <Stop offset="100%" stopColor={GLOW_COLORS.green} />
          </LinearGradient>
        </Defs>

        <Circle
          cx={size / 2}
          cy={size / 2}
          r={radius}
          stroke="url(#ringGradient)"
          strokeWidth={strokeWidth}
          fill="transparent"
          strokeLinecap="round"
        />
      </Svg>

      <View
        style={[
          styles.innerRing,
          {
            width: size - 28,
            height: size - 28,
            borderRadius: (size - 28) / 2,
          },
        ]}
      />
    </AnimatedView>
  );
}

const styles = StyleSheet.create({
  wrapper: {
    justifyContent: "center",
    alignItems: "center",
  },
  outerGlow: {
    position: "absolute",
    backgroundColor: "transparent",
    shadowOpacity: 0.28,
    shadowRadius: 24,
  },
  innerRing: {
    position: "absolute",
    borderWidth: 1,
    borderColor: "rgba(255,255,255,0.05)",
  },
});
5. GlowParticles.tsx
import React, { useEffect, useMemo } from "react";
import { Dimensions, StyleSheet } from "react-native";
import Animated, {
  Easing,
  useAnimatedStyle,
  useSharedValue,
  withRepeat,
  withSequence,
  withTiming,
} from "react-native-reanimated";
import { GLOW_COLORS } from "./glowTokens";

const { width, height } = Dimensions.get("window");

type Particle = {
  x: number;
  y: number;
  size: number;
  color: string;
  driftX: number;
  driftY: number;
  duration: number;
};

function ParticleDot({ particle }: { particle: Particle }) {
  const tx = useSharedValue(0);
  const ty = useSharedValue(0);
  const opacity = useSharedValue(0.4);

  useEffect(() => {
    tx.value = withRepeat(
      withSequence(
        withTiming(particle.driftX, {
          duration: particle.duration,
          easing: Easing.inOut(Easing.ease),
        }),
        withTiming(-particle.driftX, {
          duration: particle.duration,
          easing: Easing.inOut(Easing.ease),
        })
      ),
      -1,
      true
    );

    ty.value = withRepeat(
      withSequence(
        withTiming(particle.driftY, {
          duration: particle.duration + 500,
          easing: Easing.inOut(Easing.ease),
        }),
        withTiming(-particle.driftY, {
          duration: particle.duration + 500,
          easing: Easing.inOut(Easing.ease),
        })
      ),
      -1,
      true
    );

    opacity.value = withRepeat(
      withSequence(
        withTiming(0.9, { duration: 1800 }),
        withTiming(0.35, { duration: 1800 })
      ),
      -1,
      true
    );
  }, [opacity, particle, tx, ty]);

  const animatedStyle = useAnimatedStyle(() => ({
    transform: [{ translateX: tx.value }, { translateY: ty.value }],
    opacity: opacity.value,
  }));

  return (
    <Animated.View
      style={[
        styles.dot,
        {
          left: particle.x,
          top: particle.y,
          width: particle.size,
          height: particle.size,
          borderRadius: particle.size / 2,
          backgroundColor: particle.color,
          shadowColor: particle.color,
        },
        animatedStyle,
      ]}
    />
  );
}

export default function GlowParticles() {
  const particles = useMemo<Particle[]>(
    () => [
      { x: width * 0.18, y: height * 0.17, size: 6, color: GLOW_COLORS.cyan, driftX: 4, driftY: 8, duration: 3200 },
      { x: width * 0.78, y: height * 0.13, size: 6, color: GLOW_COLORS.green, driftX: 5, driftY: 10, duration: 4000 },
      { x: width * 0.12, y: height * 0.42, size: 5, color: GLOW_COLORS.cyan, driftX: 6, driftY: 8, duration: 3600 },
      { x: width * 0.82, y: height * 0.48, size: 4, color: GLOW_COLORS.green, driftX: 3, driftY: 7, duration: 3400 },
      { x: width * 0.28, y: height * 0.74, size: 6, color: GLOW_COLORS.purple, driftX: 6, driftY: 10, duration: 3900 },
      { x: width * 0.70, y: height * 0.63, size: 7, color: GLOW_COLORS.green, driftX: 4, driftY: 10, duration: 3700 },
      { x: width * 0.50, y: height * 0.87, size: 6, color: GLOW_COLORS.cyan, driftX: 4, driftY: 9, duration: 4200 },
      { x: width * 0.88, y: height * 0.70, size: 6, color: GLOW_COLORS.cyan, driftX: 5, driftY: 8, duration: 3500 },
    ],
    []
  );

  return (
    <>
      {particles.map((particle, index) => (
        <ParticleDot key={index} particle={particle} />
      ))}
    </>
  );
}

const styles = StyleSheet.create({

6. GlowProgressLine.tsx
import React, { useEffect } from "react";
import { Dimensions, StyleSheet, View } from "react-native";
import Animated, {
  Easing,
  useAnimatedStyle,
  useSharedValue,
  withTiming,
} from "react-native-reanimated";
import { LinearGradient } from "expo-linear-gradient";
import { GLOW_COLORS } from "./glowTokens";

const AnimatedView = Animated.createAnimatedComponent(View);
const { width } = Dimensions.get("window");

type GlowProgressLineProps = {
  progress: number;
};

export default function GlowProgressLine({ progress }: GlowProgressLineProps) {
  const animatedProgress = useSharedValue(0);
  const barWidth = Math.min(width * 0.72, 320);

  useEffect(() => {
    animatedProgress.value = withTiming(progress, {
      duration: 700,
      easing: Easing.inOut(Easing.ease),
    });
  }, [animatedProgress, progress]);

  const fillStyle = useAnimatedStyle(() => ({
    width: barWidth * animatedProgress.value,
  }));

  const dotStyle = useAnimatedStyle(() => ({
    transform: [{ translateX: barWidth * animatedProgress.value - 8 }],
    opacity: animatedProgress.value > 0.02 ? 1 : 0,
  }));

  return (
    <View style={[styles.wrapper, { width: barWidth }]}>
      <View style={styles.baseLine} />

      <AnimatedView style={[styles.fillWrap, fillStyle]}>
        <LinearGradient
          colors={[GLOW_COLORS.purple, GLOW_COLORS.cyan, GLOW_COLORS.green]}
          start={{ x: 0, y: 0 }}
          end={{ x: 1, y: 0 }}
          style={styles.fill}
        />
      </AnimatedView>

      <AnimatedView style={[styles.dot, dotStyle]} />
    </View>
  );
}

7. GlowStatusText.tsx
import React, { useEffect, useMemo, useState } from "react";
import { StyleSheet, Text, View } from "react-native";
import Animated, {
  FadeIn,
  FadeOut,
  Layout,
} from "react-native-reanimated";
import { GLOW_COLORS } from "./glowTokens";

const STATUS_MESSAGES = [
  "INITIALIZING GLOW OS",
  "SYNCING PLAYER PROFILE",
  "LOADING AI COACH",
  "PREPARING PERFORMANCE DATA",
  "COURT SYSTEMS ONLINE",
];

type GlowStatusTextProps = {
  isReady: boolean;
};

export default function GlowStatusText({ isReady }: GlowStatusTextProps) {
  const [index, setIndex] = useState(0);

  const messages = useMemo(() => STATUS_MESSAGES, []);

  useEffect(() => {
    if (isReady) return;

    const interval = setInterval(() => {
      setIndex((prev) => (prev + 1) % messages.length);
    }, 850);

    return () => clearInterval(interval);
  }, [isReady, messages.length]);

  const text = isReady ? "SYSTEM READY" : messages[index];

  return (
    <View style={styles.container}>
      <Animated.Text
        key={text}
        entering={FadeIn.duration(250)}
        exiting={FadeOut.duration(200)}
        layout={Layout}
        style={styles.text}
      >
        {text}
      </Animated.Text>
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    height: 24,
    justifyContent: "center",
    alignItems: "center",
  },
  text: {
    color: GLOW_COLORS.textMuted,
    fontSize: 12,
    letterSpacing: 2.4,
    fontWeight: "600",
  },
});

  useAnimatedStyle,
  useSharedValue,
  withRepeat,
  withSequence,
  withTiming,

import GlowParticles from "./GlowParticles";
import GlowProgressLine from "./GlowProgressLine";
import GlowRing from "./GlowRing";
import GlowStatusText from "./GlowStatusText";

const { width, height } = Dimensions.get("window");
const AnimatedView = Animated.createAnimatedComponent(View);
const AnimatedText = Animated.createAnimatedComponent(Text);

type GlowLoadingScreenProps = {
  onFinish?: () => void;
  logoSource?: any; // require("...")
};

export default function GlowLoadingScreen({
  onFinish,
  logoSource,
}: GlowLoadingScreenProps) {
  const [progress, setProgress] = useState(0);
  const [isReady, setIsReady] = useState(false);

  const logoPulse = useSharedValue(1);
  const logoOpacity = useSharedValue(0);
  const titleOpacity = useSharedValue(0);
  const sweepX = useSharedValue(-220);

  useEffect(() => {
    logoOpacity.value = withTiming(1, { duration: 700 });
    titleOpacity.value = withTiming(1, { duration: 900 });

    logoPulse.value = withRepeat(
      withSequence(
        withTiming(1.035, {
          duration: 1100,
          easing: Easing.inOut(Easing.ease),
        }),
        withTiming(1, {
          duration: 1100,
          easing: Easing.inOut(Easing.ease),
        })
      ),
      -1,
      true
    );

    sweepX.value = withRepeat(
      withSequence(
        withTiming(220, { duration: 850, easing: Easing.out(Easing.ease) }),
        withTiming(-220, { duration: 2200, easing: Easing.linear })
      ),
      -1,
      false
    );
  }, [logoOpacity, logoPulse, sweepX, titleOpacity]);

  useEffect(() => {
    const totalDuration = 3800;
    const stepTime = 220;
    const increment = stepTime / totalDuration;

    const interval = setInterval(() => {
      setProgress((prev) => {
        const next = Math.min(prev + increment, 1);

        if (next >= 1) {
          clearInterval(interval);
          setIsReady(true);

          setTimeout(() => {
            onFinish?.();
          }, 550);
        }

        return next;
      });
    }, stepTime);

    return () => clearInterval(interval);
  }, [onFinish]);

  const logoAnimatedStyle = useAnimatedStyle(() => ({
    transform: [{ scale: logoPulse.value }],
    opacity: logoOpacity.value,
  }));

  const titleAnimatedStyle = useAnimatedStyle(() => ({
    opacity: titleOpacity.value,
  }));

  const sweepStyle = useAnimatedStyle(() => ({
    transform: [{ translateX: sweepX.value }, { rotate: "0deg" }],
  }));

  const progressPercent = useMemo(() => Math.round(progress * 100), [progress]);

  return (
    <View style={styles.container}>
      <LinearGradient
        colors={[GLOW_COLORS.bg0, GLOW_COLORS.bg1, GLOW_COLORS.bg2, GLOW_COLORS.bg0]}
        locations={[0, 0.45, 0.78, 1]}
        style={StyleSheet.absoluteFill}
      />

      <View style={styles.centerBloom} />
      <GlowParticles />

      <View style={styles.content}>
        <View style={styles.heroBlock}>
          <GlowRing size={width * 0.68} strokeWidth={4} />

          <AnimatedView style={[styles.logoWrap, logoAnimatedStyle]}>
            {logoSource ? (
              <Image source={logoSource} style={styles.logoImage} resizeMode="contain" />
            ) : (
              <Text style={styles.guFallback}>GU</Text>
            )}
          </AnimatedView>

          <AnimatedView style={[styles.wordmarkWrap, titleAnimatedStyle]}>
            <Text style={styles.glowUp}>
              <Text style={styles.glowText}>GLOW </Text>
              <Text style={styles.upText}>UP</Text>
            </Text>
            <Text style={styles.sports}>SPORTS</Text>
            <Text style={styles.tagline}>GLOW UP YOUR GAME</Text>

            <View style={styles.sweepMask}>
              <AnimatedView style={[styles.lightSweep, sweepStyle]} />
            </View>
          </AnimatedView>
        </View>

        <View style={styles.bottomBlock}>
          <GlowStatusText isReady={isReady} />

          <View style={styles.progressWrap}>
            <GlowProgressLine progress={progress} />
            <Text style={styles.percent}>{progressPercent}%</Text>
          </View>

          <AnimatedText entering={FadeIn.duration(400)} style={styles.subStatus}>
            {isReady ? "READY TO PERFORM" : "BOOTING COURT SYSTEMS"}
          </AnimatedText>
        </View>
      </View>
    </View>
  );
}
